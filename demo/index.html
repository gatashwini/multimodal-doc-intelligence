01 — System Design

Pipeline Architecture

Two-pass ingestion merges Tesseract OCR signal with GPT-4o visual descriptions into hybrid semantic vectors, enabling retrieval across text, tables, and diagrams.

// Ingestion signals combined per chunk
  • OCR text: raw characters, numbers, dates
  • Visual desc: table structure, chart data, layout
  • Metadata: source_file, page_number, doc_type
  • has_visual flag for visual-dominant pages
// Retrieval: hybrid BM25 + dense (α=0.6)
  • Dense: OpenAI text-embedding-3-small
  • Sparse: BM25 keyword matching
  • Filter by doc_type (invoice / report / generic)
  • Deduplication by (file, page, chunk_index)
// Answer generation
  • GPT-4o with grounded system prompt
  • Source attribution: [Source N, Page X]
  • Low temperature (0.1) for factual extraction
  • FastAPI endpoint with latency tracking

# Two-pass ingestion: OCR + GPT-4o Vision
def ingest_pdf(pdf_path: str, doc_type: str = "generic") -> dict: pages = pdf_to_images(pdf_path) for page_num, image in pages:      # Pass 1 — pytesseract OCR ocr_text = ocr_page(image)      # Pass 2 — GPT-4o page-level visual description visual_desc = vision_describe_page(image)      # Combine both signals → chunk → upsert to Weaviate combined = f"[OCR TEXT]\n{ocr_text}\n\n[VISUAL]\n{visual_desc}" chunks = smart_chunk_text(combined, chunk_size=512, overlap=64) for chunk in chunks: collection.data.insert({ "content": chunk, "source_file": path.name, "page_number": page_num, "doc_type": doc_type, "has_visual": bool(visual_desc), })
# Hybrid search: BM25 + dense (α=0.6)
def hybrid_search( query: str, top_k: int = 6, alpha: float = 0.6, # 0=BM25, 1=dense doc_type_filter: str | None = None, ) -> list[RetrievedChunk]:
results = collection.query.hybrid( query=query, alpha=alpha, # favors semantic over keyword limit=top_k, filters=filters, return_metadata=MetadataQuery(score=True), ) return [ RetrievedChunk( content=obj.properties["content"], source_file=obj.properties["source_file"], page_number=obj.properties["page_number"], score=obj.metadata.score, ) for obj in results.objects ]
# RAG chain: retrieve → generate → attribute sources
SYSTEM_PROMPT = """You are a precise document analyst. Rules: 1. Answer ONLY from context. Never hallucinate. 2. Cite sources as [Source N, Page X] inline. 3. If not found: 'I could not find this information.' 4. Be concise. Prefer bullets for multi-part answers."""
def answer_question(question: str, top_k: int = 6) -> QAResponse: chunks = hybrid_search(question, top_k=top_k) chunks = deduplicate_chunks(chunks) context = build_context(chunks)
response = client.chat.completions.create( model="gpt-4o", messages=[ {"role": "system", "content": SYSTEM_PROMPT}, {"role": "user", "content": f"Context:\n{context}\n\nQ: {question}"}, ], temperature=0.1, # low temp for factual extraction ) return QAResponse(answer=response.choices[0].message.content, sources=chunks)
